var Singleton = (function () {
    var instance;
    var instance1;
    var instance2;

    function createInstance() {
        var object = new Object("I am a Christian T. Bellita");
        return object;
    }   
    function createInstance1() {
        var object = new Object("I am a 4th Year student");
        return object;
    }  
    function createInstance2() {
        var object = new Object("Holding the course of Bachelor of Science in Information Technology");
        return object;
    }   

    return {
        getInstance: function () {
            if (!instance) {
                instance = createInstance();
            }
        },
        getInstance1: function(){
            if (!instance1) {
                instance1 = createInstance1();
            }
            return instance1;
        },
        getInstance2: function(){
            if (!instance2) {
                instance2 = createInstance2();
            }
            return instance2;
        }
    };
})();
run();
function run() {

    var instance1 = Singleton.getInstance();
    var instance2 = Singleton.getInstance();
    var instance3 = Singleton.getInstance1();
    var instance4 = Singleton.getInstance1();
    var instance5 = Singleton.getInstance2();
    var instance6 = Singleton.getInstance2();

    console.log("Same Student? " + (instance1 === instance2));
    console.log("Same Year? " + (instance3 === instance4));
    console.log("Same Course? " + (instance5 === instance6));
}
